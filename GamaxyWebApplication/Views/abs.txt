        [HttpPost]
        public async Task<IActionResult> Wishgames(int? id)
        {
            if (HttpContext.Session.GetInt32("userid") != null)
            {
                var userId = Convert.ToInt32(HttpContext.Session.GetInt32("userid"));

                var prod = await _context.Games.Include(p => p.Category).FirstOrDefaultAsync(p => p.GameId == id);

                if (prod == null || prod.Category == null)
                {
                    return NotFound("Game or category not found.");
                }

                // Validate cateId
                var categoryExists = await _context.Categories.AnyAsync(c => c.CategoryId == prod.Category.CategoryId);
                if (!categoryExists)
                {
                    return BadRequest("Invalid category.");
                }

                var existingWishlistItem = await _context.Favourites
                    .FirstOrDefaultAsync(w => w.gameId == prod.GameId && w.UserId == userId);

                if (existingWishlistItem == null)
                {
                    var addToWishlist = new Favourite()
                    {
                        gameId = prod.GameId,
                        cateId = prod.Category.CategoryId, 
                        UserId = userId
                    };

                    await _context.Favourites.AddAsync(addToWishlist);
                    await _context.SaveChangesAsync();
                }

                return RedirectToAction("Favourite ");
            }
            return RedirectToAction("Login");
        }
